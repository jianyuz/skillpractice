            // Difficulty: Easy

class Solution:
    def isPalindrome(self, x: int) -> bool:
        if x < 0 : 
            return False
        if x == 0:
            return True

        #x is positive
        digits = []
        while x%10 > 0 or x//10>0:
            digit = x % 10
            x = x//10
            digits.append(digit)
        print(digits)
        
        num_digits = len(digits)
        for i in range(num_digits//2):
            if digits[i] != digits[num_digits - i -1]:
                return False
        return True
